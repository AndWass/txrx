Examples

Basic usage

use txrx::{just, SenderExt};
use either::Either;
let just_2 = just(1).and_then(|x| {
    if x % 2 == 0 {
        Either::Left(just(format!("{} is even", x)))
    }
    else {
        Either::Right(just(x+1))
    }
}).sync_wait().unwrap();
assert_eq!(just_2.unwrap_right(), 2);
